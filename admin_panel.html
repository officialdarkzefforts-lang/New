⚙️ SETTINGS
Firebase Configuration

To connect this admin panel to your Firebase project:
1. Open both user_panel.html and admin_panel.html
2. Replace YOUR_API_KEY and other placeholders with your Firebase config
3. In Firebase Console → Authentication → Enable Email/Password & Google sign-in
4. In Firestore → Create collections: topics, users
5. Set window.ADMIN_EMAIL to your admin email in admin_panel.html

Firestore Rules
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /topics/{topicId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null 
        && request.auth.token.email == "YOUR_ADMIN_EMAIL";
    }
    match /users/{userId} {
      allow read, write: if request.auth != null 
        && request.auth.uid == userId;
      allow read, write: if request.auth != null 
        && request.auth.token.email == "YOUR_ADMIN_EMAIL";
    }
  }
}
